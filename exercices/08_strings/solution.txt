SOLUTION 1 : Longueur d'une string
```c
#include <stdio.h>
#include <string.h>

int main() {
    char texte[100] = "Bonjour tout le monde";

    int longueur = strlen(texte);
    printf("Texte : %s\n", texte);
    printf("Longueur : %d caracteres\n", longueur);

    return 0;
}
```

SOLUTION 3 : Comparer deux strings
```c
#include <stdio.h>
#include <string.h>

int main() {
    char mot1[50] = "chat";
    char mot2[50] = "chien";

    printf("Mot 1 : %s\n", mot1);
    printf("Mot 2 : %s\n", mot2);

    int resultat = strcmp(mot1, mot2);

    if (resultat == 0) {
        printf("Les mots sont identiques\n");
    } else if (resultat < 0) {
        printf("\"%s\" vient avant \"%s\" dans l'ordre alphabetique\n", mot1, mot2);
    } else {
        printf("\"%s\" vient apres \"%s\" dans l'ordre alphabetique\n", mot1, mot2);
    }

    return 0;
}
```

SOLUTION 4 : Concaténation
```c
#include <stdio.h>
#include <string.h>

int main() {
    char str1[100] = "Bonjour ";
    char str2[50] = "le monde";

    printf("Avant concatenation :\n");
    printf("str1 : %s\n", str1);
    printf("str2 : %s\n", str2);

    strcat(str1, str2);  // Ajoute str2 à la fin de str1

    printf("\nApres concatenation :\n");
    printf("str1 : %s\n", str1);

    return 0;
}
```

SOLUTION 5 : Compter les voyelles
```c
#include <stdio.h>
#include <string.h>

int main() {
    char texte[100] = "Hello World";
    int compteur = 0;

    printf("Texte : %s\n", texte);

    for (int i = 0; texte[i] != '\0'; i++) {
        char c = texte[i];
        // Convertir en minuscule pour faciliter la comparaison
        if (c >= 'A' && c <= 'Z') {
            c = c + 32;
        }

        // Vérifier si c'est une voyelle
        if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
            compteur++;
        }
    }

    printf("Nombre de voyelles : %d\n", compteur);

    return 0;
}
```

SOLUTION 6 : Inverser une string
```c
#include <stdio.h>
#include <string.h>

int main() {
    char texte[100] = "hello";
    int longueur = strlen(texte);

    printf("Original : %s\n", texte);

    // Inverser en échangeant les caractères
    for (int i = 0; i < longueur / 2; i++) {
        char temp = texte[i];
        texte[i] = texte[longueur - 1 - i];
        texte[longueur - 1 - i] = temp;
    }

    printf("Inverse : %s\n", texte);

    return 0;
}
```

SOLUTION 7 : Convertir en majuscules
```c
#include <stdio.h>
#include <string.h>

int main() {
    char texte[100] = "hello world";

    printf("Original : %s\n", texte);

    for (int i = 0; texte[i] != '\0'; i++) {
        if (texte[i] >= 'a' && texte[i] <= 'z') {
            texte[i] = texte[i] - 32;  // Différence ASCII entre minuscule et majuscule
        }
    }

    printf("Majuscules : %s\n", texte);

    return 0;
}
```

SOLUTION 8 : Palindrome
```c
#include <stdio.h>
#include <string.h>

int main() {
    char texte[100] = "radar";
    int longueur = strlen(texte);
    int est_palindrome = 1;  // 1 = vrai, 0 = faux

    printf("Texte : %s\n", texte);

    // Comparer les caractères du début et de la fin
    for (int i = 0; i < longueur / 2; i++) {
        if (texte[i] != texte[longueur - 1 - i]) {
            est_palindrome = 0;
            break;
        }
    }

    if (est_palindrome) {
        printf("\"%s\" est un palindrome\n", texte);
    } else {
        printf("\"%s\" n'est pas un palindrome\n", texte);
    }

    return 0;
}
```
